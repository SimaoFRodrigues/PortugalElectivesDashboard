// Carregar os dados limpos
rawData = FileAttachment("data/resultados.json").json()

// Lista de chaves que não são partidos para excluir da visualização
const nonPartyKeys = [
  "Votantes", "Abstenção", "Brancos", "Nulos", "Votos Val. Exp. (VVE)"
];

// Transformar os dados para o formato hierárquico que o D3.js necessita
// Estrutura: { name: "Portugal", children: [ { name: "Distrito", children: [ { name: "Partido", value: Votos } ] } ] }
data = (() => {
  const children = Object.entries(rawData).map(([districtName, districtData]) => {
    const partyChildren = Object.entries(districtData.data)
      .filter(([key, value]) => !nonPartyKeys.includes(key) && value && value["Número"] != null)
      .map(([partyName, partyData]) => ({
        name: partyName,
        value: partyData["Número"]
      }));
    
    return {
      name: districtName,
      children: partyChildren
    };
  });

  return { name: "Portugal", children: children };
})()


chart = {
  // Especificar as dimensões do gráfico
  const width = 928;
  const height = width;
  const radius = width / 6;

  // Criar a escala de cores. As cores serão baseadas nos distritos.
  const color = d3.scaleOrdinal(d3.quantize(d3.interpolateRainbow, data.children.length + 1));

  // Calcular o layout da hierarquia
  const hierarchy = d3.hierarchy(data)
      .sum(d => d.value)
      .sort((a, b) => b.value - a.value);
  const root = d3.partition()
      .size([2 * Math.PI, hierarchy.height + 1])
    (hierarchy);
  root.each(d => d.current = d);

  // Criar o gerador de arcos
  const arc = d3.arc()
      .startAngle(d => d.x0)
      .endAngle(d => d.x1)
      .padAngle(d => Math.min((d.x1 - d.x0) / 2, 0.005))
      .padRadius(radius * 1.5)
      .innerRadius(d => d.y0 * radius)
      .outerRadius(d => Math.max(d.y0 * radius, d.y1 * radius - 1));

  // Criar o contentor SVG
  const svg = d3.create("svg")
      .attr("viewBox", [-width / 2, -height / 2, width, width])
      .style("font", "10px sans-serif");

  // Adicionar os arcos
  const path = svg.append("g")
    .selectAll("path")
    .data(root.descendants().slice(1))
    .join("path")
      // A cor é definida pelo ancestral do topo (o distrito)
      .attr("fill", d => { while (d.depth > 1) d = d.parent; return color(d.data.name); })
      .attr("fill-opacity", d => arcVisible(d.current) ? (d.children ? 0.8 : 0.5) : 0)
      .attr("pointer-events", d => arcVisible(d.current) ? "auto" : "none")
      .attr("d", d => arc(d.current));

  // Tornar os arcos clicáveis se tiverem filhos (para fazer zoom)
  path.filter(d => d.children)
      .style("cursor", "pointer")
      .on("click", clicked);

  const format = d3.format(",d");
  // Adicionar o tooltip que mostra a hierarquia e os votos
  path.append("title")
      .text(d => `${d.ancestors().map(d => d.data.name).reverse().join(" > ")}\n${format(d.value)} votos`);

  // Adicionar as etiquetas de texto
  const label = svg.append("g")
      .attr("pointer-events", "none")
      .attr("text-anchor", "middle")
      .style("user-select", "none")
    .selectAll("text")
    .data(root.descendants().slice(1))
    .join("text")
      .attr("dy", "0.35em")
      .attr("fill-opacity", d => +labelVisible(d.current))
      .attr("transform", d => labelTransform(d.current))
      .text(d => d.data.name);

  // Círculo central para voltar ao zoom inicial
  const parent = svg.append("circle")
      .datum(root)
      .attr("r", radius)
      .attr("fill", "none")
      .attr("pointer-events", "all")
      .on("click", clicked);

  // Função para lidar com o zoom ao clicar
  function clicked(event, p) {
    parent.datum(p.parent || root);

    root.each(d => d.target = {
      x0: Math.max(0, Math.min(1, (d.x0 - p.x0) / (p.x1 - p.x0))) * 2 * Math.PI,
      x1: Math.max(0, Math.min(1, (d.x1 - p.x0) / (p.x1 - p.x0))) * 2 * Math.PI,
      y0: Math.max(0, d.y0 - p.depth),
      y1: Math.max(0, d.y1 - p.depth)
    });

    const t = svg.transition().duration(750);

    path.transition(t)
        .tween("data", d => {
          const i = d3.interpolate(d.current, d.target);
          return t => d.current = i(t);
        })
      .filter(function(d) {
        return +this.getAttribute("fill-opacity") || arcVisible(d.target);
      })
        .attr("fill-opacity", d => arcVisible(d.target) ? (d.children ? 0.8 : 0.5) : 0)
        .attr("pointer-events", d => arcVisible(d.target) ? "auto" : "none") 
        .attrTween("d", d => () => arc(d.current));

    label.filter(function(d) {
        return +this.getAttribute("fill-opacity") || labelVisible(d.target);
      }).transition(t)
        .attr("fill-opacity", d => +labelVisible(d.target))
        .attrTween("transform", d => () => labelTransform(d.current));
  }
  
  function arcVisible(d) {
    return d.y1 <= 3 && d.y0 >= 1 && d.x1 > d.x0;
  }

  function labelVisible(d) {
    return d.y1 <= 3 && d.y0 >= 1 && (d.y1 - d.y0) * (d.x1 - d.x0) > 0.03;
  }

  function labelTransform(d) {
    const x = (d.x0 + d.x1) / 2 * 180 / Math.PI;
    const y = (d.y0 + d.y1) / 2 * radius;
    return `rotate(${x - 90}) translate(${y},0) rotate(${x < 180 ? 0 : 180})`;
  }

  return svg.node();
}
